### setup commands:
```
yarn add react react-dom @types/react @types/react-dom
yarn add --dev webpack webpack-cli webpack-dev-server ts-loader react-scripts typescript
```

```
yarn webpack build --mode development
yarn webpack serve --mode development
```

```
npm install --save-dev react-scripts typescript ts-loader webpack webpack-cli webpack-dev-server
npm install react react-dom @types/react @types/react-dom
```
